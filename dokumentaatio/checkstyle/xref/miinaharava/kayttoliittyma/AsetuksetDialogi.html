25      /**
26       * Metodi, joka asettelee dialogin näkyville ja palauttaa lopuksi tiedot
27       * siitä mitä asetuksia käyttäjä on tehnyt.
28       *
29       * @return kentän ominaisuudet, int[leveys, korkeus, miinat]
30       */
31      public int[] naytaDialogi() {
32          this.setLocationRelativeTo(null);
33          asetaAktiivisiksi(false);
34          this.setVisible(true);
35          return keraaTiedotPalautettavaksi();
36      }
37  
38      /**
39       * Metodi, joka listaa palautettavat tiedot kentän asetuksisa.
40       *
41       * @return kentän ominaisuudet, int[leveys, korkeus, miinat]
42       */
43      private int[] keraaTiedotPalautettavaksi() {
44          if (jRadioButton1.isSelected()) {
45              return new int[]{9, 9, 10};    // palauttaa asetukset helppoon vaikeustasoon
46          } else if (jRadioButton2.isSelected()) {
47              return new int[]{16, 16, 40};  // palauttaa asetukset keskivaikeaan vaikeustasoon
48          } else if (jRadioButton3.isSelected()) {
49              return new int[]{30, 16, 99};  // palauttaa asetukset vaikeaan vaikeustasoon
50          } else if (jRadioButton4.isSelected()) {
51              return customAsetukset();      // palauttaa asetukset custom vaikeustasoon
52          }
53          return new int[]{0, 0, 0};         // virhe palautetaan vain nollaa jolloin muutksia ei tehdä   
54      }
55  
56      /**
57       * Metodi, joka kerää custom asetuksiin tarvittavat tiedot
58       *
59       * @return kentän ominaisuudet, int[leveys, korkeus, miinat]
60       */
61      private int[] customAsetukset() {
62          int leveys = (Integer) this.jSpinner1.getValue();
63          int korkeus = (Integer) this.jSpinner2.getValue();
64          int miinat = (Integer) this.jSpinner3.getValue();
65          return new int[]{leveys, korkeus, miinat};
66      }

68      /**
69       * Metodi, joka asettelee custom asetuksia varten tarvittavat
70       * käyttöliittymän komponentit aktiivisiksi tai passiivisiksi parametrina
71       * saadun boolean arvon mukaan.
72       *
73       * @param arvo Asetetaanko aktiiviseksi
74       */
75      private void asetaAktiivisiksi(boolean arvo) {
76          this.jSpinner1.setEnabled(arvo);
77          this.jSpinner2.setEnabled(arvo);
78          this.jSpinner3.setEnabled(arvo);
79      }
