23  public class Kayttoliittyma implements Runnable {
24  
25      private JFrame frame;
26  
27      public Kayttoliittyma() {
28      }
29  
30      @Override
31      public void run() {
32          frame = new JFrame("Miinaharava");
33          frame.setPreferredSize(new Dimension((9 + 1) * 40, (9 + 1) * 40 + 100));
34          frame.setResizable(false);
35  
36          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
37  
38          JMenuBar menuBar = new JMenuBar();
39          JMenu peliMenu = new JMenu("Peli");
40          JMenuItem asetukset = new JMenuItem("Asetukset");
41          asetukset.addActionListener(new ValikkonappienKuuntelija(asetukset));
42          peliMenu.add(asetukset);
43          peliMenu.add(new JSeparator());
44          JMenuItem lopeta = new JMenuItem("Lopeta");
45          lopeta.addActionListener(new ValikkonappienKuuntelija(lopeta));
46          peliMenu.add(lopeta);
47          JMenu ohjeMenu = new JMenu("Ohje");
48          JMenuItem saannot = new JMenuItem("Säännöt");
49          saannot.addActionListener(new ValikkonappienKuuntelija(saannot));
50          ohjeMenu.add(saannot);
51          menuBar.add(peliMenu);
52          menuBar.add(ohjeMenu);
53          frame.setJMenuBar(menuBar);
54  
55          luoKomponentit(frame.getContentPane());
56  
57          frame.pack();
58          frame.setLocationRelativeTo(null);
59          frame.setVisible(true);
60      }
61  
62      /**
63       * Metodi luo käyttöliittymän komponentit ja asettelee ne paikoilleen.
64       *
65       * @param container : container-olio, johon käyttöliittymän komponentit
66       * asetellaan
67       */
68      private void luoKomponentit(Container container) {
69          container.setLayout(new BorderLayout());
70  
71          Ruudukko ruudukko = new Ruudukko(9, 9, 10);
72          Pelikentta kentta = new Pelikentta(ruudukko);
73          container.add(kentta);
74  
75          JPanel paneeli2 = new JPanel(new GridLayout(1, 3));
76          paneeli2.setBackground(Color.red);
77          paneeli2.setPreferredSize(new Dimension(500, 50));
78          container.add(paneeli2, BorderLayout.NORTH);
79  
80          JButton nappi1 = new JButton();
81          paneeli2.add(new JPanel());
82          paneeli2.add(nappi1);
83          paneeli2.add(new JPanel());
84          nappi1.setText("Aloita alusta");
85  
86      }
87  
88      public JFrame getFrame() {
89          return frame;
90      }
