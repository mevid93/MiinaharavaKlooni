1   package miinaharava.kuuntelijat;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import miinaharava.kayttoliittyma.Kayttoliittyma;
6   
7   /**
8    * Luokka tarjoaa metodin käyttöliittymässä sijaitsevalle "aloita-alusta"-napin
9    * toiminnalle.
10   *
11   * @author Martin Vidjeskog
12   */
13  public class AloitusnapinKuuntelija implements ActionListener {
14  
15      private Kayttoliittyma ikkuna;  // käyttöliittymä ikkuna olio
16  
17      /**
18       * Konstruktori metodi, joka luo uuden AloitusnapinKuuntelijan ja tallentaa
19       * muistiin parametrina annetun Kayttoliittyma-olion.
20       *
21       * @param ikkuna Kayttoliittyma-olio
22       */
23      public AloitusnapinKuuntelija(Kayttoliittyma ikkuna) {
24          this.ikkuna = ikkuna;
25      }
26  
27      /**
28       * Metodi, joka toteutetaan kun tapahtuma rekisteröidään napissa, johon
29       * kuuntelija on kytketty. Kutsutaan käyttöliittymä-olion metodia joka
30       * asettaa pelin alkamaan alusta.
31       *
32       * @param e Napin painallus tapahtuma
33       */
34      @Override
35      public void actionPerformed(ActionEvent e) {
36          this.ikkuna.aloitaAlusta();
37      }
38  
39  }
