1   package miinaharava.domain;
2   
3   import java.awt.Color;
4   import java.awt.Dimension;
5   import java.awt.GridBagConstraints;
6   import java.awt.GridBagLayout;
7   import java.awt.Insets;
8   import javax.swing.JPanel;
9   import miinaharava.domain.Ruudukko;
10  import miinaharava.kuuntelijat.RuutunapinKuuntelija;
11  
12  /**
13   * Luokka, joka kuvaa miinaharavan pelikenttää. Luokka tietää pelikentän koon ja
14   * sisältää taulukon kaikista ruuduista (napeista), jotka pelikenttään kuuluvat.
15   *
16   * @author Martin Vidjeskog
17   */
18  public class Pelikentta extends JPanel {
19  
20      private int leveys;
21      private int korkeus;
22      private Ruudukko ruudukko;
23  
24      /**
25       * Konstruktori, joka luo uuden Pelikentta-olion ja saa parametrina
26       * Ruudukko-olion.
27       *
28       * @param ruudukko Ruudukko olio
29       */
30      public Pelikentta(Ruudukko ruudukko) {
31          this.leveys = ruudukko.getLeveys();
32          this.korkeus = ruudukko.getKorkeus();
33          this.ruudukko = ruudukko;
34          this.asetteleRuutunapit();
35      }
36  
37      /**
38       * Metodi, joka kutsuu toista metodia ruutunappien luomiseksi ja asettelee
39       * saamansa ruudut jPaneeliin.
40       */
41      private void asetteleRuutunapit() {
42          this.setLayout(new GridBagLayout());
43          this.setBackground(Color.GRAY);
44          GridBagConstraints c = new GridBagConstraints();
45          c.fill = GridBagConstraints.BOTH;
46          for (int i = 0; i < this.korkeus; i++) {
47              c.gridy = i;
48              for (int j = 0; j < this.leveys; j++) {
49                  c.gridx = j;
50                  Ruutunappi ruutunappi = luoRuutunappi(j, i);
51                  this.add(ruutunappi, c);
52              }
53          }
54      }
55  
56      /**
57       * Metodi, joka luo yksittäisen Ruutunappi-olion ja antaa sille
58       * parameteinaan saadut koordinaatit.
59       *
60       * @param x Koordinaatti pelikentässä
61       * @param y Koordinaatti pelikentässä
62       *
63       * @return luotu Ruutunappi-olio
64       */
65      private Ruutunappi luoRuutunappi(int x, int y) {
66          Ruutunappi ruutunappi = new Ruutunappi(x, y);
67          ruutunappi.setPreferredSize(new Dimension(40, 40));
68          ruutunappi.addMouseListener(new RuutunapinKuuntelija(ruutunappi, this.ruudukko));
69          ruutunappi.setMargin(new Insets(0, 0, 0, 0));
70          return ruutunappi;
71      }
72  
73      /**
74       * Metodi palauttaa pelikentän leveyden.
75       *
76       * @return Kentän leveys
77       */
78      public int getKentanLeveys() {
79          return leveys;
80      }
81  
82      /**
83       * Metodi palauttaa pelikentän korkeuden.
84       *
85       * @return Kentän korkeus
86       */
87      public int getKentanKorkeus() {
88          return korkeus;
89      }
90  
91  }
