1   package miinaharava.domain;
2   
3   import java.awt.Color;
4   import java.awt.Dimension;
5   import java.awt.GridBagConstraints;
6   import java.awt.GridBagLayout;
7   import java.awt.Insets;
8   import java.util.ArrayList;
9   import javax.swing.JPanel;
10  import miinaharava.kuuntelijat.RuutunapinKuuntelija;
11  
12  /**
13   * Luokka, joka kuvaa miinaharavan pelikenttää. Luokka tietää pelikentän koon ja
14   * sisältää taulukon kaikista ruuduista (napeista), jotka pelikenttään kuuluvat.
15   *
16   * @author Martin Vidjeskog
17   */
18  public class Pelikentta extends JPanel {
19  
20      private Ruudukko ruudukko;
21      private ArrayList<Ruutunappi> nappilista;
22  
23      /**
24       * Konstruktori, joka luo uuden Pelikentta-olion ja saa parametrina
25       * Ruudukko-olion.
26       *
27       * @param ruudukko Ruudukko olio
28       */
29      public Pelikentta(Ruudukko ruudukko) {
30          this.ruudukko = ruudukko;
31          this.nappilista = new ArrayList<>();
32          this.asetteleRuutunapit();
33      }
34  
35      /**
36       * Metodi, joka kutsuu toista metodia ruutunappien luomiseksi ja asettelee
37       * saamansa ruudut jPaneeliin.
38       */
39      private void asetteleRuutunapit() {
40          this.setLayout(new GridBagLayout());
41          this.setBackground(Color.GRAY);
42          GridBagConstraints c = new GridBagConstraints();
43          c.fill = GridBagConstraints.BOTH;
44          for (int i = 0; i < this.ruudukko.getKorkeus(); i++) {
45              c.gridy = i;
46              for (int j = 0; j < this.ruudukko.getLeveys(); j++) {
47                  c.gridx = j;
48                  Ruutunappi ruutunappi = luoRuutunappi(j, i);
49                  this.add(ruutunappi, c);
50              }
51          }
52      }
53  
54      /**
55       * Metodi, joka luo yksittäisen Ruutunappi-olion ja antaa sille
56       * parameteinaan saadut koordinaatit.
57       *
58       * @param x Koordinaatti pelikentässä
59       * @param y Koordinaatti pelikentässä
60       *
61       * @return luotu Ruutunappi-olio
62       */
63      private Ruutunappi luoRuutunappi(int x, int y) {
64          Ruutunappi ruutunappi = new Ruutunappi(x, y);
65          ruutunappi.setPreferredSize(new Dimension(40, 40));
66          ruutunappi.addMouseListener(new RuutunapinKuuntelija(ruutunappi, this, this.ruudukko));
67          ruutunappi.setMargin(new Insets(0, 0, 0, 0));
68          nappilista.add(ruutunappi);
69          return ruutunappi;
70      }
71  
72      /**
73       * Metodi, jolla ykittäisen Pelikentässä sijaitsevan Ruutunapin miinat
74       * iconin sellaiseksi, että se vastaa naapurimiinojen määrää.
75       *
76       * @param x Napin x-koordinaatti
77       * @param y Napin y-koordinaatti
78       * @param miinat Ruudun naapurimiinojen lkm
79       */
80      public void asetaRuutunapinNaapurimiinatEsille(int x, int y, int miinat) {
81          for (Ruutunappi nappi : this.nappilista) {
82              if (nappi.getxKoordinaatti() == x && nappi.getyKoordinaatti() == y) {
83                  nappi.naytanaapurimiinojenMaara(miinat);
84                  break;
85              }
86          }
87      }
88  
89      /**
90       * Metodi, joka asettaa kaikkien Pelikentän Ruutunappien iconiksi miinan,
91       * mikäli vastaava Ruutu sisältää miinan ja Ruutua ei ole liputettu. Metodia
92       * kutsutaan kun pelaaja on vahingossa osunut miinaan.
93       */
94      public void paljastaKaikkiMiinat() {
95          for (int i = 0; i < this.ruudukko.getKorkeus(); i++) {
96              for (int j = 0; j < this.ruudukko.getLeveys(); j++) {
97                  if (!ruudukko.onkoAvattu(j, i) && !this.ruudukko.onkoLiputettu(j, i)) {
98                      if (ruudukko.onkoMiina(j, i)) {
99                          for (Ruutunappi nappi : this.nappilista) {
100                             if (nappi.getxKoordinaatti() == j && nappi.getyKoordinaatti() == i) {
101                                 nappi.naytaMiina();
102                                 break;
103                             }
104                         }
105                     }
106                 }
107             }
108         }
109     }
110 
111     /**
112      * Metodi palauttaa Pelikentän leveyden.
113      *
114      * @return Kentän leveys
115      */
116     public int getKentanLeveys() {
117         return this.ruudukko.getLeveys();
118     }
119 
120     /**
121      * Metodi palauttaa Pelikentän korkeuden.
122      *
123      * @return Kentän korkeus
124      */
125     public int getKentanKorkeus() {
126         return this.ruudukko.getKorkeus();
127     }
128 
129     /**
130      * Metodi palauttaa Pelikentässä olevien Ruutunappien listan.
131      *
132      * @return Lista kentällä olevista Ruutunapeista
133      */
134     public ArrayList<Ruutunappi> getNappilista() {
135         return this.nappilista;
136     }
137 
138 }
