1   package miinaharava.domain;
2   
3   import java.awt.Color;
4   import javax.swing.BorderFactory;
5   import javax.swing.JButton;
6   
7   /**
8    * Luokka, joka kuvaa yksittäistä pelikenttään kuuluvaa ruutua (JButton).
9    * Sisältää toiminnot napin ulkomuodon muuttamiseen ja metodit koordinaattien
10   * saamiseen.
11   *
12   * @author Martin Vidjeskog
13   */
14  public class Ruutunappi extends JButton {
15  
16      private final int xKoordinaatti;
17      private final int yKoordinaatti;
18  
19      /**
20       * Kostruktori-metodi joka luo uuden Ruutunappi-olion ja asettelee
21       * parametrit oliomuuttujiin.
22       *
23       * @param x Ruudun x-koordinaatti pelikentässä
24       * @param y Ruudun y-koordinaatti pelikentässä
25       */
26      public Ruutunappi(int x, int y) {
27          this.xKoordinaatti = x;
28          this.yKoordinaatti = y;
29      }
30  
31      /**
32       * Metodi, joka palauttaa Ruutunapin sijainnin pelikentässä ja ruudukossa.
33       *
34       * @return x-koordinaatti pelikentässä
35       */
36      public int getxKoordinaatti() {
37          return xKoordinaatti;
38      }
39  
40      /**
41       * Metodi, joka palauttaa Ruutunapin sijainnin pelikentässä ja ruudukossa.
42       *
43       * @return y-koordinaatti pelikentässä
44       */
45      public int getyKoordinaatti() {
46          return yKoordinaatti;
47      }
48  
49      /**
50       * Metodi, joka muuttaa ruutunapin ulkonäköä vastaamaan ruutua joka pitää
51       * sisällään miinan.
52       */
53      public void naytaMiina() {
54          this.muutaPainetuksi();
55          this.setText("X");
56      }
57  
58      /**
59       * Metodi, joka muuttaa ruutunapin ulkonäköä vastaamaan avattua ruutua jossa
60       * ei ole miinaa.
61       *
62       * @param naapurimiinat Naapurimiinojen lukumäärä
63       */
64      public void naytanaapurimiinojenMaara(int naapurimiinat) {
65          this.muutaPainetuksi();
66          if (naapurimiinat != 0) {
67              this.setText("" + naapurimiinat);
68          }
69      }
70  
71      /**
72       * Metodi, joka muuttaa ruutunapin ulkonäköä vastaamaan ruutua, jossa
73       * joko on lippu tai ei ole lippua, riippuen parametrin arvosta.
74       * 
75       * @param esilla Merkitäänkö lippu esille 
76       */
77      public void naytaLippu(boolean esilla) {
78          if (esilla) {
79              this.setText("L");
80          } else {
81              this.setText("");
82          }
83      }
84  
85      /**
86       * Meotodi, jolla napin tausta muutetaan vastaamaan avatun 
87       * ruudun ulkonäköä.
88       */
89      private void muutaPainetuksi() {
90          this.setBorder(BorderFactory.createLoweredBevelBorder());
91          this.setBackground(Color.white);
92          this.setEnabled(false);
93      }
94  }
